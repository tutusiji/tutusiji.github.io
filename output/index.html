vue与react生成前端二维码的QRcode组件使用

VUE版本：

安装:

npm install qrcode

import QRCode from 'qrcode'

DOM:

<canvas id="canvas"></canvas>

页面调用

this.useqrcode('http://www.baidu.com')

useqrcode(url) {
    const canvas = document.getElementById('canvas')
    QRCode.toCanvas(
        canvas,
        url,
        {
            scale: 5.0,
            height: 180,
            wight: 180
        },
        error => {
            if (error) console.error(error)
            console.log('success!')
        }
    )
}
